import math

def equacao_segundo_grau(a, b, c):
    delta = b**2 - 4*a*c

    if delta < 0:
        print("Não existem raízes reais")

    elif delta == 0:
        x = -b / (2*a)
        print("Raiz:", x)

    else:
        x1 = (-b + math.sqrt(delta)) / (2*a)
        x2 = (-b - math.sqrt(delta)) / (2*a)

        print("x1 =", x1)
        print("x2 =", x2)

equacao_segundo_grau(1, -3, 2)
